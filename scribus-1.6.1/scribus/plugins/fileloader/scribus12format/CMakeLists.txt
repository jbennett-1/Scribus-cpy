include_directories(
	${CMAKE_SOURCE_DIR}
	${CMAKE_SOURCE_DIR}/scribus
)

set(SCR12FORMAT_FL_PLUGIN_SOURCES
	scribus12format.cpp
	scribus12formatimpl.cpp
)

set(SCRIBUS_SCR12FORMAT_FL_PLUGIN "scribus12format")

add_library(${SCRIBUS_SCR12FORMAT_FL_PLUGIN} MODULE ${SCR12FORMAT_FL_PLUGIN_SOURCES})

target_link_libraries(${SCRIBUS_SCR12FORMAT_FL_PLUGIN} ${EXE_NAME})

if(WANT_PCH)
	target_precompile_headers(${SCRIBUS_SCR12FORMAT_FL_PLUGIN} PRIVATE "../../plugins_pch.h")
endif()


install(TARGETS ${SCRIBUS_SCR12FORMAT_FL_PLUGIN}
	LIBRARY
	DESTINATION ${PLUGINDIR}
	PERMISSIONS ${PLUGIN_PERMISSIONS}
)
